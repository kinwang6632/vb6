/*
  說明: 建立EBT 之使用者帳號

  檔名: Add_User.sql
  Date: 2004.03.12
*/

DROP USER "EBTUSER";

CREATE USER "EBTUSER"  PROFILE "DEFAULT" IDENTIFIED BY "ebt" 
    DEFAULT 
    TABLESPACE "REPORT" TEMPORARY 
    TABLESPACE "REPORT_TEMP" ACCOUNT UNLOCK;


GRANT "CONNECT" TO "EBTUSER";
ALTER USER "EBTUSER"  QUOTA UNLIMITED ON REPORT;


/*
以下用系統台之使用者帳號來執行
*/

GRANT SELECT ON "SO001" TO "EBTUSER";
GRANT SELECT ON "SO002" TO "EBTUSER";
GRANT SELECT ON "SO003" TO "EBTUSER";
GRANT SELECT ON "SO014" TO "EBTUSER";
GRANT SELECT ON "SO509A" TO "EBTUSER";
GRANT SELECT ON "EMC_EBT001" TO "EBTUSER";
